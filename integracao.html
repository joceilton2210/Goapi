PHP (Backend)

function.php
<?php

function enviarWhatsApp($numero, $mensagem) {
    $curl = curl_init();

    $dados = [
        "number" => $numero,
        "message" => $mensagem
    ];

    curl_setopt_array($curl, [
        CURLOPT_URL => "http://localhost:4000/api/messages/MINHA_INSTANCIA/send-text",
        CURLOPT_RETURNTRANSFER => true,
        CURLOPT_CUSTOMREQUEST => "POST",
        CURLOPT_POSTFIELDS => json_encode($dados),
        CURLOPT_HTTPHEADER => [
            "x-api-key: change-this-key",
            "Content-Type: application/json"
        ],
    ]);

    $response = curl_exec($curl);
    curl_close($curl);
    
    return $response;
}
?>

app.js
const axios = require('axios');

async function sendMsg() {
  try {
    const response = await axios.post(
      'http://localhost:4000/api/messages/MINHA_INSTANCIA/send-text',
      {
        number: '5511999999999',
        message: 'Olá via Node.js!'
      },
      {
        headers: {
          'x-api-key': 'change-this-key'
        }
      }
    );
    console.log(response.data);
  } catch (error) {
    console.error(error);
  }
}
